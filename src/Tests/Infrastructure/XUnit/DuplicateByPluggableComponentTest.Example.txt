using Compze.Tessaging.Hosting.Testing;
using Xunit;

namespace Compze.Tests.Infrastructure.XUnit.Examples;

/// <summary>
/// Example of how to use DuplicateByPluggableComponentTest with XUnit.
/// This pattern allows you to test all pluggable component combinations defined in TestUsingPluggableComponentCombinations file.
/// </summary>
public class ExamplePluggableComponentTest : DuplicateByPluggableComponentTest
{
   /// <summary>
   /// Example test that runs once for each pluggable component combination.
   /// The PluggableComponentTestContext is automatically injected with the parsed combination.
   /// </summary>
   [PluggableComponentsTheory]
   public void Example_test_that_runs_for_all_combinations(PluggableComponentTestContext context)
   {
      // Access the parsed values directly from the context object
      var persistenceLayer = context.PersistenceLayer;
      var diContainer = context.DIContainer;

      // You also have access to the combination string if needed
      System.Console.WriteLine($"Testing with: {context.Combination}");

      // Use ValueFor to get different values per persistence layer
      var someValue = context.ValueFor<string>(
         msSql: "SQL Server specific value",
         mySql: "MySQL specific value",
         pgSql: "PostgreSQL specific value",
         memory: "In-memory specific value"
      );

      // Your test logic here
      Assert.NotEqual(default, persistenceLayer);
      Assert.NotEqual(default, diContainer);
      Assert.NotNull(someValue);
   }

   /// <summary>
   /// Example of a regular test that doesn't use pluggable components.
   /// This will run only once.
   /// </summary>
   [Fact]
   public void Example_regular_test()
   {
      // Regular test logic
      Assert.True(true);
   }
}
